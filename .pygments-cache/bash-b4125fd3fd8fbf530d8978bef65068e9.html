# First we need to create a key pair for the new certificate
openssl genrsa -out host.key 2048</p>

<h1 id="then-we-will-use-the-key-pair-to-generate-a-certificate-signing-request-or-csr">Then we will use the key pair to generate a Certificate Signing Request or CSR</h1>
<p># This is what you send to valid certificate authorities to ask them to create &amp; sign a valid certificate for you
openssl req -new -key host.key -out host.csr</p>

<h1 id="now-we-can-create-a-valid-certificate-and-sign-it-with-our-rootca">Now we can create a valid certificate and sign it with our rootCA</h1>
<p>openssl x509 -req -in host.csr -CA rootCA.crt -CAkey rootCA.key -CAcreateserial -out host.crt -days 365
