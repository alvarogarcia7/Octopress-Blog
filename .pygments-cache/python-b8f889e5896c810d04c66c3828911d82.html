def xor(mydata,mykey):
    keylen = len(mykey)
    datalen = len(mydata)
    
    # easier to just extend the key array, but probably not that efficient
    # not that we care about it here ;)
    key = mykey * ( (datalen/keylen)+1 ) 
    
    return ''.join(chr(ord(a) ^ ord(b)) for a,b in zip(mydata,key))
